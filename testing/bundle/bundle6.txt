### FILE: test_programs/testIf_var_var.imp
PROGRAM IS
  a,b,c
IN
  READ a;
  READ b;
  IF a >  b THEN c := 1; WRITE c; ENDIF
  IF a <  b THEN c := 2; WRITE c; ENDIF
  IF a >= b THEN c := 3; WRITE c; ENDIF
  IF a <= b THEN c := 4; WRITE c; ENDIF
  IF a =  b THEN c := 5; WRITE c; ENDIF
  IF a != b THEN c := 6; WRITE c; ENDIF
  WRITE a;
  WRITE b;
END
